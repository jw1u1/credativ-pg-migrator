	p_value          VARCHAR(255),

	p_error_text     VARCHAR(255) out)
RETURNS RECORD AS $$
DECLARE
    locvar_vorhanden VARCHAR(10);
    locvar_unbekannt VARCHAR(10);
    locvar_ausdruecklicheEinwilligungTelefonisch VARCHAR(10);
    locvar_voiceFileValue VARCHAR(255);
    locvar_voiceFileType NUMERIC(10,0);
BEGIN

	locvar_vorhanden := 'vorhanden';
	locvar_unbekannt := 'unbekannt';
	locvar_ausdruecklicheEinwilligungTelefonisch := '10';
	locvar_voiceFileType := 124;

	p_voiceFileId := null;

    if(p_value = locvar_ausdruecklicheEinwilligungTelefonisch and (p_propertyTypeId in (117, 118, 119, 120))) then
	    begin

        select value INTO locvar_voiceFileValue
		from rel_customer_customerproperty
        where customer_id = p_customer_id
        and customerproperty_id in (select customerproperty_id
                                    from customerproperty
                                    where customerpropertytype_id = locvar_voiceFileType);

        Exception when others THEN
		begin
	        p_error_code := 600;
	        p_error_text  = 'getVoiceFile: Error selecting voice file information';
	        RETURN 1;
		end;

		end;

        if(lower(locvar_voiceFileValue) = 'ja') then
			p_voiceFileId := locvar_vorhanden;
		else
    		p_voiceFileId := locvar_unbekannt;
		end if;

    ELSE
	begin
		p_voiceFileId := null;
	end;
	end if;

	RETURN 0;
END;
